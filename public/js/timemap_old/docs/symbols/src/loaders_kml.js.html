 15 /**
 16  * @class
 17  * KML loader factory - inherits from remote loader
 18  *
 19  * <p>This is a loader class for KML files. Currently supports all geometry
 20  * types (point, polyline, polygon, and overlay) and multiple geometries.</p>
 21  *
 22  * @example Usage in TimeMap.init():
 23  
 24     datasets: [
 25         {
 26             title: "KML Dataset",
 27             type: "kml",
 28             options: {
 29                 url: "mydata.kml"   // Must be local
 30             }
 31         }
 32     ]
 33  *
 34  * @param {Object} options          All options for the loader:<pre>
 35  *   {Array} url                        URL of KML file to load (NB: must be local address)
 36  *   {Function} preloadFunction         Function to call on data before loading
 37  *   {Function} transformFunction       Function to call on individual items before loading
 38  * </pre>
 39  * @return {TimeMap.loaders.remote} Remote loader configured for KML
 40  */
 41 TimeMap.loaders.kml = function(options) {
 42     var loader = new TimeMap.loaders.remote(options);
 43     loader.parse = TimeMap.loaders.kml.parse;
 44     return loader;
 45 }
 46 
 47 /**
 48  * Static function to parse KML with time data.
 49  *
 50  * @param {XML string} kml      KML to be parsed
 51  * @return {TimeMapItem Array}  Array of TimeMapItems
 52  */
 53 TimeMap.loaders.kml.parse = function(kml) {
 54     var items = [], data, kmlnode, placemarks, pm, i, j;
 55     kmlnode = GXml.parse(kml);
 56     
 57     // get TimeMap utilty functions
 58     // assigning to variables should compress better
 59     var util = TimeMap.util;
 60     var getTagValue = util.getTagValue,
 61         getNodeList = util.getNodeList,
 62         makePoint = util.makePoint,
 63         makePoly = util.makePoly,
 64         formatDate = util.formatDate;
 65     
 66     // recursive time data search
 67     var findNodeTime = function(n, data) {
 68         var check = false;
 69         // look for instant timestamp
 70         var nList = getNodeList(n, "TimeStamp");
 71         if (nList.length > 0) {
 72             data.start = getTagValue(nList[0], "when");
 73             check = true;
 74         }
 75         // otherwise look for span
 76         else {
 77             nList = getNodeList(n, "TimeSpan");
 78             if (nList.length > 0) {
 79                 data.start = getTagValue(nList[0], "begin");
 80                 data.end = getTagValue(nList[0], "end") ||
 81                     // unbounded spans end at the present time
 82                     formatDate(new Date());
 83                 check = true;
 84             }
 85         }
 86         // try looking recursively at parent nodes
 87         if (!check) {
 88             var pn = n.parentNode;
 89             if (pn.nodeName == "Folder" || pn.nodeName=="Document") {
 90                 findNodeTime(pn, data);
 91             }
 92             pn = null;
 93         }
 94     };
 95     
 96     // look for placemarks
 97     placemarks = getNodeList(kmlnode, "Placemark");
 98     for (i=0; i<placemarks.length; i++) {
 99         pm = placemarks[i];
100         data = { options: {} };
101         // get title & description
102         data.title = getTagValue(pm, "name");
103         data.options.description = getTagValue(pm, "description");
104         // get time information
105         findNodeTime(pm, data);
106         // find placemark(s)
107         var nList, coords, pmobj;
108         data.placemarks = [];
109         // look for marker
110         nList = getNodeList(pm, "Point");
111         for (j=0; j<nList.length; j++) {
112             pmobj = { point: {} };
113             // get lat/lon
114             coords = getTagValue(nList[j], "coordinates");
115             pmobj.point = makePoint(coords, 1);
116             data.placemarks.push(pmobj);
117         }
118         // look for polylines
119         nList = getNodeList(pm, "LineString");
120         for (j=0; j<nList.length; j++) {
121             pmobj = { polyline: [] };
122             // get lat/lon
123             coords = getTagValue(nList[j], "coordinates");
124             pmobj.polyline = makePoly(coords, 1);
125             data.placemarks.push(pmobj);
126         }
127         // look for polygons
128         nList = getNodeList(pm, "Polygon");
129         for (j=0; j<nList.length; j++) {
130             pmobj = { polygon: [] };
131             // get lat/lon
132             coords = getTagValue(nList[j], "coordinates");
133             pmobj.polygon = makePoly(coords, 1);
134             // XXX: worth closing unclosed polygons?
135             data.placemarks.push(pmobj);
136         }
137         items.push(data);
138     }
139     
140     // look for ground overlays
141     placemarks = getNodeList(kmlnode, "GroundOverlay");
142     for (i=0; i<placemarks.length; i++) {
143         pm = placemarks[i];
144         data = { options: {}, overlay: {} };
145         // get title & description
146         data.title = getTagValue(pm, "name");
147         data.options.description = getTagValue(pm, "description");
148         // get time information
149         findNodeTime(pm, data);
150         // get image
151         nList = getNodeList(pm, "Icon");
152         data.overlay.image = getTagValue(nList[0], "href");
153         // get coordinates
154         nList = getNodeList(pm, "LatLonBox");
155         data.overlay.north = getTagValue(nList[0], "north");
156         data.overlay.south = getTagValue(nList[0], "south");
157         data.overlay.east = getTagValue(nList[0], "east");
158         data.overlay.west = getTagValue(nList[0], "west");
159         items.push(data);
160     }
161     
162     // clean up
163     kmlnode = null;
164     placemarks = null;
165     pm = null;
166     nList = null;
167     return items;
168 };
